      select: { credits: true }
    });

  async consumeCredits(userId: string, amount: number, reason: string): Promise<boolean> {
    const user = await prisma.user.findUnique({
      where: { id: userId },
      select: { credits: true }

      return false;

      prisma.user.update({
        where: { id: userId },
        data: { credits: { decrement: amount } }
      }),

        data: {
          userId,
          amount: -amount,
          reason
        }
      })
    ]);
    
    return true;
  }

    await prisma.$transaction([
      prisma.user.update({

      }),
